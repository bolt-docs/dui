import { Terminal, Paintbrush, ListOrdered, Table, MessageSquare, Activity, Cog, Type, PanelTop, SquareDashed, SplitSquareHorizontal, Bot, TrendingUp, Play, Palette, Package, File, BarChart3, GitCompare, QrCode, Image as ImageIcon } from 'lucide-react'

**DUI** — terminal UI toolkit for Node.js CLI apps. ANSI-aware, zero-dependency (just `string-width`), tree-shakeable.

---

## Quick Start

```bash
pnpm add @bdocs/dui
```

```ts
import { configure, info, success, box, bullet } from '@bdocs/dui'

configure({ prefix: 'mytool' })

info('Starting build...')
success('Build completed!')

console.log(box(['Output ready at dist/'], { title: 'Result', style: 'round' }))
console.log(bullet(['File A', 'File B', 'File C']))
```
